def _register_colormaps() -> None:
    """Register custom colormaps for heatmaps."""
    import matplotlib as mpl

    cmaps = {
        "editorial_diverging": LinearSegmentedColormap.from_list(
            "editorial_diverging", [DOWN_COLOR, "#1a1a1c", UP_COLOR], N=256,
        ),
        "editorial_warm": LinearSegmentedColormap.from_list(
            "editorial_warm",
            [PAPER, "#3a3530", "#7a5e52", DOWN_COLOR, ACCENT], N=256,
        ),
        "editorial_cool": LinearSegmentedColormap.from_list(
            "editorial_cool",
            [PAPER, "#2a2a2c", "#6e6a64", "#a8a59d", FG_COLOR], N=256,
        ),
    }
    for name, cmap in cmaps.items():
        try:
            mpl.colormaps.register(cmap, name=name, force=True)
        except (AttributeError, TypeError):
            mpl.cm.register_cmap(name=name, cmap=cmap)


def add_crisis_shading(ax: plt.Axes, alpha: float = 0.08) -> None:
    """Add vertical shading for crisis periods to a time-series axes."""
    for label, (start, end) in CRISIS_PERIODS.items():
        ax.axvspan(start, end, alpha=alpha, color=DOWN_COLOR, zorder=0)


def add_is_oos_divider(ax: plt.Axes) -> None:
    """Add a vertical dashed line at the IS/OOS split date."""
    ax.axvline(IS_OOS_SPLIT, color=ACCENT, linewidth=0.8, linestyle="--",
               alpha=0.6, zorder=1)


def savefig(fig: plt.Figure, name: str) -> Path:
    """Save figure to FIGURES_DIR at 300 DPI and close it."""
    path = FIGURES_DIR / name
    fig.savefig(path, dpi=DPI, bbox_inches="tight", pad_inches=0.2)
    plt.close(fig)
    logger.info("Saved %s", path)
    return path

"""Performance visualization: cumulative returns, drawdown, heatmap, rolling Sharpe.

Charts: 01_cumulative_returns, 02_drawdown, 03_monthly_heatmap,
        04_rolling_sharpe, 11_cost_sensitivity, 12_is_oos_comparison,

import logging
from pathlib import Path

import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import numpy as np
import pandas as pd
import seaborn as sns

from commodity_curve_factors.utils.paths import DATA_DIR, RESULTS_DIR
from commodity_curve_factors.evaluation.metrics import (

from commodity_curve_factors.evaluation.attribution import rolling_sharpe
from commodity_curve_factors.visualization.style import (

    STRATEGY_COLORS, BENCHMARK_COLORS, STRATEGY_LABELS,

logger = logging.getLogger(__name__)

BACKTEST_DIR = DATA_DIR / "processed" / "backtest"


def _load_strategies() -> dict[str, pd.DataFrame]:
    """Load all strategy backtest results."""
    strategies = {}
    for f in sorted(BACKTEST_DIR.glob("*.parquet")):
        if f.stem in ("benchmarks", "cost_sensitivity"):
            continue
        strategies[f.stem] = pd.read_parquet(f)
    return strategies


def _load_benchmarks() -> pd.DataFrame:
    """Load benchmark returns."""
    return pd.read_parquet(BACKTEST_DIR / "benchmarks.parquet")


def _load_cost_sensitivity() -> pd.DataFrame:
    """Load cost sensitivity sweep results."""
    return pd.read_parquet(BACKTEST_DIR / "cost_sensitivity.parquet")


def _load_performance_summary() -> pd.DataFrame:
    """Load pre-computed performance summary table."""
    return pd.read_parquet(RESULTS_DIR / "tables" / "performance_summary.parquet")


def _get_color(name: str) -> str:
    """Get color for a strategy or benchmark."""
    return STRATEGY_COLORS.get(name, BENCHMARK_COLORS.get(name, "#eae6de40"))


def _get_label(name: str) -> str:
    """Get display label for a strategy or benchmark."""
    return STRATEGY_LABELS.get(name, name.replace("_", " ").title())


def plot_cumulative_returns() -> Path:
    """Chart 01: Cumulative returns for all strategies + benchmarks."""
    strategies = _load_strategies()
    benchmarks = _load_benchmarks()

    fig, ax = plt.subplots(figsize=(14, 6))

    # Plot benchmarks first (behind)
    for col in benchmarks.columns:
        if col.endswith("_cumulative"):
            name = col.replace("_cumulative", "")
            ax.plot(benchmarks.index, benchmarks[col],
                    color=_get_color(name), linewidth=0.8,
                    label=_get_label(name), alpha=0.7)

    # Plot strategies
    hero_strategies = ["tsmom", "tsi"]
    for name, df in strategies.items():
        if "cumulative" not in df.columns:
            continue
        lw = 2.0 if name in hero_strategies else 0.8
        ax.plot(df.index, df["cumulative"],
                color=_get_color(name), linewidth=lw,
                label=_get_label(name))

    add_crisis_shading(ax)
    add_is_oos_divider(ax)

    ax.set_ylabel("Cumulative Return")
    ax.set_title("Strategy Performance (2005–2024)")
    ax.legend(loc="upper left", fontsize=7, ncol=2)
    ax.grid(True, axis="y")

    return savefig(fig, "01_cumulative_returns.png")


def plot_drawdown() -> Path:
    """Chart 02: Drawdown comparison for key strategies."""
    strategies = _load_strategies()
    benchmarks = _load_benchmarks()

    show = ["tsmom", "tsi", "xs_carry", "multi_factor_ew"]
    fig, ax = plt.subplots(figsize=(14, 6))

    # EW long benchmark drawdown
    if "equal_weight_long_cumulative" in benchmarks.columns:
        cum = benchmarks["equal_weight_long_cumulative"]
        dd = cum / cum.cummax() - 1
        ax.fill_between(dd.index, dd, 0, alpha=0.3,
                        color=BENCHMARK_COLORS["equal_weight_long"],
                        label="EW Long")

    for name in show:
        if name not in strategies:
            continue
        df = strategies[name]
        if "drawdown" in df.columns:
            ax.fill_between(df.index, df["drawdown"], 0, alpha=0.25,
                            color=_get_color(name), label=_get_label(name))

    add_crisis_shading(ax, alpha=0.05)
    ax.set_ylabel("Drawdown")
    ax.set_title("Drawdown from Peak")
    ax.legend(loc="lower left", fontsize=8)
    ax.grid(True, axis="y")

    return savefig(fig, "02_drawdown.png")


def plot_monthly_heatmap() -> Path:
    """Chart 03: Monthly returns heatmap for EW Long benchmark."""
    benchmarks = _load_benchmarks()

    # Use equal_weight_long returns
    col = [c for c in benchmarks.columns if "equal_weight_long" in c
           and "return" in c.lower()]
    if not col:
        # Fallback: compute from cumulative
        cum = benchmarks.get("equal_weight_long_cumulative")
        if cum is None:
            logger.warning("No EW Long data for monthly heatmap")
            return FIGURES_DIR / "03_monthly_heatmap.png"
        returns = cum.pct_change()
    else:
        returns = benchmarks[col[0]]

    monthly = returns.resample("ME").apply(lambda x: (1 + x).prod() - 1)
    pivot = pd.DataFrame({
        "year": monthly.index.year,
        "month": monthly.index.month,
        "return": monthly.values,
    }).pivot(index="year", columns="month", values="return")
    pivot.columns = ["Jan", "Feb", "Mar", "Apr", "May", "Jun",
                     "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"]

    fig, ax = plt.subplots(figsize=(12, 8))
    sns.heatmap(pivot, annot=True, fmt=".1%", center=0,
                cmap="editorial_diverging", linewidths=0.5,
                linecolor=BG_COLOR, cbar_kws={"shrink": 0.8},
                ax=ax)
    ax.set_title("Equal-Weight Long — Monthly Returns")
    ax.set_ylabel("")

    return savefig(fig, "03_monthly_heatmap.png")


def plot_rolling_sharpe() -> Path:
    """Chart 04: 252-day rolling Sharpe for key strategies."""
    strategies = _load_strategies()
    benchmarks = _load_benchmarks()

    fig, ax = plt.subplots(figsize=(14, 6))

    show = {"tsi": "TSI", "tsmom": "TSMOM"}
    for name, label in show.items():
        if name in strategies and "net_return" in strategies[name].columns:
            rs = rolling_sharpe(strategies[name]["net_return"], window=252)
            ax.plot(rs.index, rs, color=_get_color(name), linewidth=1.4,
                    label=label)

    # EW Long rolling Sharpe
    ew_col = [c for c in benchmarks.columns if "equal_weight_long" in c
              and "return" in c.lower()]
    if ew_col:
        rs_ew = rolling_sharpe(benchmarks[ew_col[0]], window=252)
        ax.plot(rs_ew.index, rs_ew,
                color=BENCHMARK_COLORS["equal_weight_long"],
                linewidth=1.0, label="EW Long")

    add_is_oos_divider(ax)
    ax.axhline(0, color=FG_COLOR, linewidth=0.5, alpha=0.3)
    ax.set_ylabel("Rolling Sharpe (252d)")
    ax.set_title("Rolling Sharpe Ratio")
    ax.legend(loc="upper left", fontsize=9)
    ax.grid(True, axis="y")

    return savefig(fig, "04_rolling_sharpe.png")


def plot_cost_sensitivity() -> Path:

    if "cost_bps" in df.columns and "sharpe" in df.columns:

def plot_is_oos_comparison() -> Path:
    """Chart 12: IS vs OOS Sharpe side-by-side grouped bars."""
    summary = _load_performance_summary()

    fig, ax = plt.subplots(figsize=(12, 6))

    strategies = summary["strategy"].values
    x = np.arange(len(strategies))
    width = 0.35

    is_sharpe = summary["is_sharpe"].values
    oos_sharpe = summary["oos_sharpe"].values

    bars_is = ax.bar(x - width / 2, is_sharpe, width, label="In-Sample",
                     color=ACCENT, alpha=0.8)
    bars_oos = ax.bar(x + width / 2, oos_sharpe, width, label="Out-of-Sample",
                      color=DOWN_COLOR, alpha=0.8)

    ax.set_xticks(x)
    ax.set_xticklabels([_get_label(s) for s in strategies],
                       rotation=45, ha="right", fontsize=8)
    ax.axhline(0, color=FG_COLOR, linewidth=0.5, alpha=0.3)
    ax.set_ylabel("Sharpe Ratio")
    ax.set_title("In-Sample vs Out-of-Sample Sharpe")
    ax.legend(fontsize=9)
    ax.grid(True, axis="y")

    return savefig(fig, "12_is_oos_comparison.png")


def plot_performance_table() -> Path:
    """Chart 15: Performance summary rendered as a figure."""
    summary = _load_performance_summary()

    cols = ["strategy", "full_sharpe", "is_sharpe", "oos_sharpe",
            "is_cagr", "oos_cagr", "is_max_dd", "oos_max_dd"]
    available = [c for c in cols if c in summary.columns]
    table_data = summary[available].copy()
    table_data["strategy"] = table_data["strategy"].map(
        lambda s: _get_label(s)
    )

    fig, ax = plt.subplots(figsize=(14, 8))
    ax.axis("off")

    # Format numeric columns
    for col in available:
        if col == "strategy":
            continue
        table_data[col] = table_data[col].apply(
            lambda v: f"{v:+.2f}" if pd.notna(v) else "—"
        )

    headers = [c.replace("_", " ").title() for c in available]
    cell_text = table_data.values.tolist()

    table = ax.table(
        cellText=cell_text,
        colLabels=headers,
        cellLoc="center",
        loc="center",
    )
    table.auto_set_font_size(False)
    table.set_fontsize(9)
    table.scale(1, 1.6)

    # Style table
    for key, cell in table.get_celld().items():
        cell.set_edgecolor("#eae6de1e")
        if key[0] == 0:
            cell.set_facecolor("#1a1a1e")
            cell.set_text_props(color=FG_COLOR, fontweight="bold")
        else:
            cell.set_facecolor(PAPER)
            cell.set_text_props(color="#eae6de8c")

    ax.set_title("Performance Summary", pad=20, fontsize=14)

    return savefig(fig, "15_performance_table.png")


def generate_all() -> list[Path]:
    """Generate all performance charts."""
    paths = []
    for fn in [plot_cumulative_returns, plot_drawdown, plot_monthly_heatmap,
               plot_rolling_sharpe, plot_cost_sensitivity,
               plot_is_oos_comparison, plot_performance_table]:
        try:
            paths.append(fn())
        except Exception:
            logger.exception("Failed to generate %s", fn.__name__)
    return paths

import logging
from pathlib import Path

import pandas as pd

from commodity_curve_factors.utils.paths import DATA_DIR

def _compute_layer_returns() -> dict[str, pd.Series]:
    """Reconstruct per-layer returns from saved signals and prices.

    Recomputes the three layers independently rather than modifying
    the backtest module to save intermediates.
    """
    # Load front-month returns
    futures = load_futures()
    returns = pd.DataFrame()
    for sym, df in futures.items():
        col = "close" if "close" in df.columns else "Close"
        if col in df.columns:
            returns[sym] = df[col].pct_change()
    returns = returns.dropna(how="all")

    # Load curves and compute CY
    curves = {}
    for f in sorted(CURVES_DIR.glob("*.parquet")):
        curves[f.stem] = pd.read_parquet(f)

    macro_path = DATA_DIR / "raw" / "macro"
    rf_path = macro_path / "DGS3MO.parquet"
    rf = pd.read_parquet(rf_path).squeeze() if rf_path.exists() else pd.Series(
        2.0, index=returns.index
    )

    storage = estimate_storage_cost(curves)
    daily_cy = compute_convenience_yield(curves, rf, storage)
    monthly_cy = monthly_convenience_yield(daily_cy)

    # Load TSMOM
    tsmom_path = FACTORS_DIR / "tsmom.parquet"
    tsmom = pd.read_parquet(tsmom_path) if tsmom_path.exists() else pd.DataFrame(
        0, index=returns.index, columns=returns.columns,
    )

    strategy_cfg = load_config("strategy")
    tsi_cfg = strategy_cfg.get("tsi", {})

    layers = {}

    # Layer 1: Directional
    try:
        dir_cfg = tsi_cfg.get("curve_directional", {})
        l1_weights = build_directional_weights(
            monthly_cy, tsmom, returns.index,
            thresholds=dir_cfg.get("regime_thresholds"),
            position_map=dir_cfg.get("position_map"),
            trend_up_mult=dir_cfg.get("trend_up_mult", 1.2),
            trend_down_mult=dir_cfg.get("trend_down_mult", 0.7),
        )
        common = l1_weights.columns.intersection(returns.columns)
        l1_ret = (l1_weights[common] * returns[common]).sum(axis=1)
        layers["Layer 1: Directional"] = l1_ret
    except Exception:
        logger.warning("Could not compute Layer 1 returns")

    # Layer 2: Transition
    try:
        trans_cfg = tsi_cfg.get("curve_transition", {})
        transition = compute_transition_signal(
            monthly_cy, lookback=trans_cfg.get("lookback_days", 63),
        )
        l2_weights = transition_to_position(
            transition, tsmom, threshold=trans_cfg.get("threshold_std", 0.5),
        )
        common = l2_weights.columns.intersection(returns.columns)
        l2_ret = (l2_weights[common] * returns[common]).sum(axis=1)
        layers["Layer 2: Transition"] = l2_ret
    except Exception:
        logger.warning("Could not compute Layer 2 returns")

    # Layer 3: Spreads (simplified — just crack spread)
    try:
        if all(c in daily_cy.columns for c in ["CL", "RB", "HO"]):
            cy_crack = compute_cy_crack(daily_cy)
            spread_cfg = tsi_cfg.get("structural_spreads", {})
            crack_cfg = spread_cfg.get("crack_spread", {})
            l3_weights = crack_spread_signal(
                cy_crack, threshold=crack_cfg.get("z_threshold", 1.5),
            )
            common = l3_weights.columns.intersection(returns.columns)
            l3_ret = (l3_weights[common] * returns[common]).sum(axis=1)
            layers["Layer 3: Spreads"] = l3_ret
    except Exception:
        logger.warning("Could not compute Layer 3 returns")

    # Combined TSI from saved backtest
    tsi_path = BACKTEST_DIR / "tsi.parquet"
    if tsi_path.exists():
        tsi_df = pd.read_parquet(tsi_path)
        if "net_return" in tsi_df.columns:
            layers["Combined TSI"] = tsi_df["net_return"]

    return layers

    ax.axhline(0, color=FG_COLOR, linewidth=0.5, alpha=0.3)
    ax.set_ylabel("Annualised Contribution (%)")
    ax.set_title("Sector Attribution — Equal-Weight Long")
    ax.grid(True, axis="y")

    # Add value labels
    for bar, val in zip(bars, contributions):
        ax.text(bar.get_x() + bar.get_width() / 2, bar.get_height(),
                f"{val:.1f}%", ha="center", va="bottom", fontsize=9,
                color=FG_COLOR)

    return savefig(fig, "13_sector_attribution.png")


def plot_bootstrap_ci() -> Path:
    """Chart 14: Bootstrap Sharpe CI for key strategies."""
    strategies_to_plot = ["tsmom", "tsi", "xs_carry", "multi_factor_ew"]
    benchmark_name = "equal_weight_long"

    fig, ax = plt.subplots(figsize=(12, 6))

    all_names = []
    all_points = []
    all_los = []
    all_his = []
    all_colors = []

    for name in strategies_to_plot:
        path = BACKTEST_DIR / f"{name}.parquet"
        if not path.exists():
            continue
        df = pd.read_parquet(path)
        if "net_return" not in df.columns:
            continue
        try:
            point, lo, hi = bootstrap_sharpe_ci(df["net_return"])
            all_names.append(_get_label(name))
            all_points.append(point)
            all_los.append(lo)
            all_his.append(hi)
            all_colors.append(_get_color(name))
        except Exception:
            logger.warning("Bootstrap failed for %s", name)

    # Add EW Long benchmark
    benchmarks = pd.read_parquet(BACKTEST_DIR / "benchmarks.parquet")
    ew_col = [c for c in benchmarks.columns if "equal_weight_long" in c
              and "return" in c.lower()]
    if ew_col:
        try:
            point, lo, hi = bootstrap_sharpe_ci(benchmarks[ew_col[0]])
            all_names.append("EW Long")
            all_points.append(point)
            all_los.append(lo)
            all_his.append(hi)
            all_colors.append(BENCHMARK_COLORS["equal_weight_long"])
        except Exception:
            pass

    if not all_names:
        ax.text(0.5, 0.5, "No bootstrap data", transform=ax.transAxes,
                ha="center", color=FG_COLOR)
        return savefig(fig, "14_bootstrap_ci.png")

    y = np.arange(len(all_names))

    # CI bars
    for i in range(len(all_names)):
        ax.barh(y[i], all_his[i] - all_los[i], left=all_los[i],
                height=0.5, color=all_colors[i], alpha=0.3)
        ax.plot(all_points[i], y[i], "o", color=all_colors[i], markersize=8)

    ax.axvline(0, color=DOWN_COLOR, linewidth=0.8, linestyle="--", alpha=0.5)
    ax.set_yticks(y)
    ax.set_yticklabels(all_names, fontsize=10)
    ax.set_xlabel("Sharpe Ratio")
    ax.set_title("Bootstrap 95% Confidence Intervals")
    ax.grid(True, axis="x")

    return savefig(fig, "14_bootstrap_ci.png")


def generate_all() -> list[Path]:
    """Generate all risk charts."""

        try:

        except Exception:
            logger.exception("Failed to generate %s", fn.__name__)
    return paths

"""Generate all publication-quality figures and website chart data.

Usage:
    python -m commodity_curve_factors.visualization

"""

import json
import logging
from pathlib import Path

import pandas as pd

from commodity_curve_factors.utils.paths import DATA_DIR, RESULTS_DIR, PROJECT_ROOT
from commodity_curve_factors.visualization.style import setup, FIGURES_DIR

logging.basicConfig(
    level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s"
)
logger = logging.getLogger(__name__)

WEBSITE_JS_DIR = PROJECT_ROOT / "website" / "js"
WEBSITE_ASSETS_DIR = PROJECT_ROOT / "website" / "assets" / "figures"
BACKTEST_DIR = DATA_DIR / "processed" / "backtest"


def _generate_chart_data_js() -> Path:
    """Generate chart_data_inline.js for Plotly interactives on the website."""
    WEBSITE_JS_DIR.mkdir(parents=True, exist_ok=True)
    output_path = WEBSITE_JS_DIR / "chart_data_inline.js"

    # ─── CHART_DATA: equity curves ───
    chart_data = {"strategies": {}}
    for f in sorted(BACKTEST_DIR.glob("*.parquet")):
        if f.stem in ("cost_sensitivity",):
            continue
        df = pd.read_parquet(f)
        if f.stem == "benchmarks":
            # Extract each benchmark
            for col in df.columns:
                if col.endswith("_cumulative"):
                    name = col.replace("_cumulative", "")
                    chart_data["strategies"][name] = {
                        "dates": [d.strftime("%Y-%m-%d") for d in df.index],
                        "cumulative_return": [
                            round(v, 4) if pd.notna(v) else None
                            for v in df[col].values
                        ],
                    }
        elif "cumulative" in df.columns:
            chart_data["strategies"][f.stem] = {
                "dates": [d.strftime("%Y-%m-%d") for d in df.index],
                "cumulative_return": [
                    round(v, 4) if pd.notna(v) else None
                    for v in df["cumulative"].values
                ],
            }

    # ─── COST_DATA: cost sensitivity ───
    cost_path = BACKTEST_DIR / "cost_sensitivity.parquet"
    cost_data = {"bps": [], "sharpe": [], "cagr": [], "maxdd": []}
    if cost_path.exists():
        cs = pd.read_parquet(cost_path)
        for col_name in ["cost_bps", "bps"]:
            if col_name in cs.columns:
                cost_data["bps"] = cs[col_name].tolist()
                break
        for col_name in ["sharpe", "sharpe_ratio"]:
            if col_name in cs.columns:
                cost_data["sharpe"] = [round(v, 3) for v in cs[col_name]]
                break
        for col_name in ["cagr"]:
            if col_name in cs.columns:
                cost_data["cagr"] = [round(v, 3) for v in cs[col_name]]
                break
        for col_name in ["max_dd", "max_drawdown", "maxdd"]:
            if col_name in cs.columns:
                cost_data["maxdd"] = [round(v, 3) for v in cs[col_name]]
                break

    # ─── CY_DATA: convenience yield for interactive explorer ───
    cy_data = {"dates": [], "commodities": {}, "regime_labels": [
        "Deep Contango", "Mild Contango", "Balanced",
        "Mild Backwardation", "Crisis Backwardation",
    ]}
    try:
        from commodity_curve_factors.curves.convenience_yield import (
            estimate_storage_cost, compute_convenience_yield,
            monthly_convenience_yield,
        )
        from commodity_curve_factors.signals.curve_regime import classify_regime

        curves_dir = DATA_DIR / "processed" / "curves"
        curves = {}
        for f in sorted(curves_dir.glob("*.parquet")):
            curves[f.stem] = pd.read_parquet(f)

        macro_path = DATA_DIR / "raw" / "macro" / "DGS3MO.parquet"
        rf = pd.read_parquet(macro_path).squeeze() if macro_path.exists() else (
            pd.Series(2.0, index=next(iter(curves.values())).index)
        )

        storage = estimate_storage_cost(curves)
        daily_cy = compute_convenience_yield(curves, rf, storage)
        monthly_cy = monthly_convenience_yield(daily_cy)
        regimes = classify_regime(monthly_cy)

        # Downsample daily CY to weekly for JS size
        weekly_cy = daily_cy.resample("W-FRI").last()
        cy_data["dates"] = [d.strftime("%Y-%m-%d") for d in weekly_cy.index]

        for sym in weekly_cy.columns:
            cy_vals = [
                round(v, 4) if pd.notna(v) else None
                for v in weekly_cy[sym].values
            ]
            # Map regimes to weekly
            regime_vals = []
            for d in weekly_cy.index:
                # Find most recent monthly regime
                prior = regimes.index[regimes.index <= d]
                if len(prior) > 0 and sym in regimes.columns:
                    val = regimes.loc[prior[-1], sym]
                    regime_map = {
                        "deep_contango": 0, "mild_contango": 1,
                        "balanced": 2, "mild_backwardation": 3,
                        "crisis_backwardation": 4,
                    }
                    regime_vals.append(regime_map.get(val, 2))
                else:
                    regime_vals.append(2)
            cy_data["commodities"][sym] = {
                "cy": cy_vals, "regime": regime_vals,
            }
    except Exception:
        logger.exception("Failed to generate CY data for website")

    # Write JS file
    js_content = (
        f"var CHART_DATA = {json.dumps(chart_data)};\n"
        f"var COST_DATA = {json.dumps(cost_data)};\n"
        f"var CY_DATA = {json.dumps(cy_data)};\n"
    )
    output_path.write_text(js_content)
    logger.info("Wrote %s (%.1f KB)", output_path,
                len(js_content) / 1024)
    return output_path


def _copy_figures_to_website() -> None:
    """Copy generated PNGs to website assets directory."""
    import shutil
    WEBSITE_ASSETS_DIR.mkdir(parents=True, exist_ok=True)
    for png in FIGURES_DIR.glob("*.png"):
        dest = WEBSITE_ASSETS_DIR / png.name
        shutil.copy2(png, dest)
    logger.info("Copied %d figures to %s",
                len(list(FIGURES_DIR.glob("*.png"))), WEBSITE_ASSETS_DIR)


def main() -> None:
    """Generate all figures and website data."""
    setup()

    from commodity_curve_factors.visualization import (
        performance, factors, curves, tsi, risk,
    )

    all_paths = []
    for module in [performance, factors, curves, tsi, risk]:
        paths = module.generate_all()
        all_paths.extend(paths)

    logger.info("Generated %d charts", len(all_paths))

    # Generate website JS data
    _generate_chart_data_js()

    # Copy figures to website
    _copy_figures_to_website()

    logger.info("Done. Charts in %s, website data in %s",
                FIGURES_DIR, WEBSITE_JS_DIR)


if __name__ == "__main__":
    main()
